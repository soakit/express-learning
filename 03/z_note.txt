        mergeParams 从父路由合并params
        caseSensitive 大小写敏感

    链式调用
        router.route('/test').get(function(req, res, next) {
            console.log('one')
            next()
        }).get(function(req, res) {
            console.log('two')
            res.send('ok')
        }).post(function(req, res, next) {
            console.log('three')
            next()
        }).get(function(req, res) {
            console.log('four')
            res.send('ok')
        })
